Dataclass Dict relations

convert DC to dict

  • asdict method: convert D.C to dict
  • unpack: init dataclass with unpack dict into init method
  • acite package: simplifies creation of data classes (PEP 557) from dictionaries.

asdict

from dataclasses import dataclass, asdict

@dataclass
class Point3D:
    x:int
    y:int
    z:int


point_3d = Point3D(1,2,3)
print(point_3d)
d_data = asdict(point_3d)
print(d_data)

# Result
dataclass: Point3D(x=1, y=2, z=3)
Dict: {'x': 1, 'y': 2, 'z': 3}

unpack

from dataclasses import dataclass, asdict

@dataclass
class Point3D:
    x:int
    y:int
    z:int

point_data = {
    "x": 1,
    "y": 2,
    "z": 3
}

point_3d = Point3D(**point_data)
print(point_data)
print(point_3d)


# Result
Dict: {'x': 1, 'y': 2, 'z': 3}
Dataclass: Point3D(x=1, y=2, z=3)

dacite

github

pip install dacite
  • Demo using nested with optional
  • More examples in github repo
from dataclasses import dataclass
from typing import Optional
from dacite import from_dict

@dataclass
class A:
    x: str
    y: Optional[int]


@dataclass
class B:
    a: A


data = {
    'a': {
        'x': 'test',
        'y': 1,
    }
}

data_opt = {
    'a': {
        'x': 'test'
    }
}

result = from_dict(data_class=B, data=data)
print(result)
result = from_dict(data_class=B, data=data_opt)
print(result)

# Result
B(a=A(x='test', y=1))
B(a=A(x='test', y=None))
